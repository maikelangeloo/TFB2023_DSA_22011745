  <label>Search for Property (Name or Price): </label>
  <input type="text" id="searchInput" placeholder="Enter name or price">

  <select id="searchType">
    <option value="linear">Linear Search</option>
    <option value="binary">Binary Search (by price only)</option>
  </select>

  <button onclick="searchProperty()">Search</button>

  </table>

  <script>
    const properties = [
      { id: 1, name: "Apartment A", price: 120000 },
      { id: 2, name: "Condo B", price: 200000 },
      { id: 3, name: "Terrace C", price: 180000 },
      { id: 4, name: "Bungalow D", price: 350000 },
      { id: 5, name: "Semi-D E", price: 280000 },
      { id: 6, name: "Flat F", price: 90000 },
    ];

    // Display property list
    const table = document.getElementById("propertyTable");
    properties.forEach(prop => {
      const row = `<tr>
        <td>${prop.id}</td>
        <td>${prop.name}</td>
        <td>${prop.price}</td>
      </tr>`;
      table.innerHTML += row;
    });

    function searchProperty() {
      const input = document.getElementById("searchInput").value.trim();
      const type = document.getElementById("searchType").value;
      const resultDiv = document.getElementById("result");

      if (!input) {
        resultDiv.innerHTML = "‚ùå Please enter a search term.";
        return;
      }

      let start = performance.now();
      let found = null;

      if (type === "linear") {
        // Linear Search by name or price
        for (let i = 0; i < properties.length; i++) {
          if (
            properties[i].name.toLowerCase() === input.toLowerCase() ||
            properties[i].price.toString() === input
          ) {
            found = properties[i];
            break;
          }
        }
      } else if (type === "binary") {
        // Binary Search by sorted price
        const sorted = [...properties].sort((a, b) => a.price - b.price);
        let low = 0, high = sorted.length - 1;
        const target = parseInt(input);
        while (low <= high) {
          const mid = Math.floor((low + high) / 2);
          if (sorted[mid].price === target) {
            found = sorted[mid];
            break;
          } else if (sorted[mid].price < target) {
            low = mid + 1;
          } else {
            high = mid - 1;
          }
        }
      }

      let end = performance.now();
      if (found) {
        resultDiv.innerHTML = `‚úÖ Found: <strong>${found.name}</strong> (RM ${found.price}) <br>‚è±Ô∏è Time taken: ${(end - start).toFixed(4)} ms`;
      } else {
        resultDiv.innerHTML = `‚ùå Property not found. <br>‚è±Ô∏è Time taken: ${(end - start).toFixed(4)} ms`;
      }
    }
  </script>
